\subsection{Euler's number $e$}
\begin{eqnarray}
	e=\lim_{n\rightarrow+\infty}\left(1+\frac{1}{n}\right)^n\\
	e=\sum_{n=0}^{\infty}\frac{1}{n!}\\
	e=2.71828\,18284\,59045\,23536\ldots\,
\end{eqnarray}

\section{Fahrenheit - Celsius}
\begin{eqnarray}
	T_C=\frac{5}{9}(T_F-32)\\
	T_F=\frac{9}{5}T_C+32
\end{eqnarray}
Thus $-40\celsius=-40\degree F$.
