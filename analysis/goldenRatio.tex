\begin{equation}
    \varphi = \frac{1+\sqrt{5}}{2}
\end{equation}

\subsection{Calculation}
\begin{eqnarray}
    \frac{a+b}{a} = \frac{a}{b} &=& \varphi\\
    1 + \frac{1}{\varphi} &=& \varphi
\end{eqnarray}

\begin{eqnarray}
    \varphi* &=& \frac{1-\sqrt{5}}{2} \\
    \frac{1}{\varphi*} &=& \varphi
\end{eqnarray}
